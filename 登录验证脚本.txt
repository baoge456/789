<script>
const REMEMBER_LOGIN = false;  // true=记住登录  false=每次都验证
const LOGIN_KEY = "is_login_success";

// ✅ 正确的  SHA256
const CORRECT_HASH = "06a9e94ceb731a81b5ed9119cd359f1883ed808cd20baa6277018b9ca3dc3975";

async function sha256(text) {
  const enc = new TextEncoder();
  const hash = await crypto.subtle.digest("SHA-256", enc.encode(text));
  return Array.from(new Uint8Array(hash)).map(b => b.toString(16).padStart(2, "0")).join("");
}

if (REMEMBER_LOGIN && localStorage.getItem(LOGIN_KEY) === "true") {
  document.getElementById("login_overlay").remove();
}

async function check() {
  const val = document.getElementById("pwd_input").value.trim();
  const inputHash = await sha256(val);
  
  if (inputHash === CORRECT_HASH) {
    if (REMEMBER_LOGIN) localStorage.setItem(LOGIN_KEY, "true");
    document.getElementById("login_overlay").remove();
  } else {
    alert("密码错误");
  }
}

document.getElementById("login_btn").onclick = check;
document.getElementById("pwd_input").onkeydown = e => { if(e.key==="Enter") check() };
setTimeout(() => document.getElementById("pwd_input").focus(), 100);
</script>
