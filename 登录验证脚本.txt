    <h3>🔒 请输入访问密码</h3>

<script>
  // ================== 配置区 ==================
  const MY_PASSWORD = "baoge888";
  const REMEMBER_LOGIN = false;  // true=记住登录  false=每次都验证
  const LOGIN_KEY = "is_login_success";

  // 打开页面时判断是否已登录
  if (REMEMBER_LOGIN && localStorage.getItem(LOGIN_KEY) === "true") {
    document.getElementById("login_overlay").remove();
  }

  // 验证逻辑
  function check() {
    let val = document.getElementById("pwd_input").value.trim();
    if (val === MY_PASSWORD) {
      // 需要记住才存状态
      if (REMEMBER_LOGIN) {
        localStorage.setItem(LOGIN_KEY, "true");
      }
      document.getElementById("login_overlay").remove();
    } else {
      alert("密码错误，请重试");
    }
  }

  // 点击登录
  document.getElementById("login_btn").onclick = check;

  // 回车登录
  document.getElementById("pwd_input").onkeydown = function (e) {
    if (e.key === "Enter") check();
  };

  // 自动聚焦
  setTimeout(() => {
    document.getElementById("pwd_input").focus();
  }, 100);
</script>
